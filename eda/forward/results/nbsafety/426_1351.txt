def plot_bound_salt(s0, b):
        dsdx = calc_dsdx(100, b, s0)
        for salt in range(30):
            bound = salt - dsdx*100
            plt.plot(salt, bound)
        ax = plt.gca()
        ax.set_xlabel('Salinity [psu]')
        ax.set_ylabel('Bound salt [psu]')
        #ax.set_xlim([0, 75])
        #ax.set_ylim([0, 50])
        plt.legend()
